Exemple carré

Objectif

Créer un carré d'étoiles de 5 caractères de côté dans la console processing

Version simple

void setup()
{
 int i= 0;
 while (i<5)
 {
  println("*****");
  i++;
 }
}

Version améliorée

void setup()
{
 int i= 0;
 int cote = 5;
 while (i<cote)
 {
  uneLigne("#", cote);
  i++;
 }
}

void uneLigne(String signe, int longueur)
{

  String ligne = "";
  int i = 0;
  while(i<longueur){
   ligne += signe; //ligne = ligne + signe
   i++;
  }
  println(ligne);
}
